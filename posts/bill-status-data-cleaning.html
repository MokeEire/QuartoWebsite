Source

This data source is derived from govinfo’s BILLSTATUS bulk data, which can be found here. The US Government Publishing Office collects and provides records of all the bills that pass through congress, including the committees which consider the bill, the legislators who sponsor, or co-sponsor the bill, and any action taken on the bill while it is in congress e.g. votes and debates. To better understand and analyze legislative decision-making institutions, I have processed and cleaned the bulk data files for the 117th congress to prepare them in nested tabular format1.

Method

Cleaning script: https://github.com/MokeEire/my-reps/blob/master/R/clean_data.R

Data processing functions: https://github.com/MokeEire/my-reps/blob/master/R/parsing_functions.R

To process the data, it was first collected in a file directory that mimics that of govinfo’s web directory e.g. ./<CONGRESS>/<BILL TYPE>/<FILE>.xml . The cleaning script and the functions used to process the data can be found above. The step-by-step process to parse each individual file is described below:

  1. Read in the XML file with xml2 and store it as an {xml_document}
  2. Extract the child elements of the <bill> node which meet the criteria below, convert the nodeset to a list object, and flatten the list object to a dataframe.

    • The element has 0 children

    • The value is not an empty string

  3. Extract the child elements of <bill> node which have at least 1 child themselves.
    • This includes committees, votes, actions, and bill (co)sponsors
  4. Convert each element from step #3 to a nodeset, process and flatten the data into a tabular form which suits the characteristic. For example:

    • Actions are processed into a dataframe which contains one row per action, with nested columns for committees and subcommittees listed on each action.

    • Votes are processed into a dataframe which contains one row per vote, with nested columns for vote roll information at the party and legislator levels.

Code

These steps are contained in a function named extract_bill_status() found at the Data processing functions link above.

extract_bill_status() function
extract_bill_status = function(xml_file, 
                               nested_attributes = c("committees", "votes", "actions", "sponsors", "cosponsors"),
                               get_votes = T,
                               col_specs = attribute_col_types,
                               log_threshold = "INFO",
                               log_types = c("file", "console")){
  # Create logger
  logger = create_logger(log_threshold = log_threshold, 
                         log_types = log_types)
  
  bill_xml = xml_child(read_xml(xml_file), "bill")
  
  # singletons = xml_singular_nodes(bill_xml)
  singletons = xml_find_all(bill_xml, "//bill/*[count(./*) = 0 and not(string-length(.) = 0)]")
  # Singletons strewn together
  bill_df = flatten_dfc(
    setNames(
      map(singletons, xml_text), 
      xml_name(singletons))
  )

  # Extract non-singular base attributes ----
  ## Policy area
  policy_areas = as_list(xml_find_all(bill_xml, "policyArea/name")) %>% 
    map_chr(flatten_chr)
  ## Subjects
  bill_subjects = as_list(xml_find_all(bill_xml, "subjects/billSubjects/legislativeSubjects/item/name")) %>% 
    map_chr(flatten_chr)
  ## Summaries
  bill_summaries = as_list(xml_find_all(bill_xml, "summaries/billSummaries/item")) %>% 
    map_dfr(flatten_dfc)
  ## Titles
  bill_titles = as_list(xml_find_all(bill_xml, "titles/item")) %>% 
    map_dfr(flatten_dfc)
  ## Text versions
  bill_text_versions = as_list(xml_find_all(bill_xml, "textVersions/item")) %>% 
    map(map_at, "formats", map_dfr, flatten_dfc) %>% 
    map_dfr(flatten_dfc)
  ## Latest action
  latest_action = as_list(xml_find_all(bill_xml, "latestAction")) %>% 
    map(flatten_dfc) %>% 
    map_dfr(~rename_with(.x, ~str_c("latestAction_", .)))
  
  bill_df$policy_areas = list(policy_areas)
  bill_df$legislative_subjects = list(bill_subjects)
  bill_df$bill_summaries = list(bill_summaries)
  bill_df$bill_titles = list(bill_titles)
  bill_df$bill_text_versions = list(bill_text_versions)
  
  bill_df = bind_cols(bill_df, latest_action)

  log_debug(logger, 
           bill_type = bill_df$billType,
           bill_num = bill_df$billNumber,
           "Reading XML")


  # Committees ---------
  log_debug(logger, 
           bill_type = bill_df$billType,
           bill_num = bill_df$billNumber,
           "Parsing committees")
  
  committees = xml_find_all(bill_xml, "//bill/committees/billCommittees/item")
  
  if("committees" %in% nested_attributes && length(committees)>0){

    # Coerce nodes to list
    committees_list = as_list(committees)
    
    committees_df = map_dfr(committees_list, parse_committee) %>% 
      janitor::clean_names()
    
    bill_df$committees = list(committees_df)
    
  } else {
    bill_df$committees = list(tibble())
  }
  
  # Votes ---------
  log_debug(logger, 
           bill_type = bill_df$billType,
           bill_num = bill_df$billNumber,
           "Parsing votes")
  
  votes_node = xml_find_all(bill_xml, "//bill/recordedVotes/recordedVote")
  
  if("votes" %in% nested_attributes && length(votes_node)>0 && get_votes){
    
    # Coerce nodes to list
    votes_list = as_list(votes_node)

    votes_df = map_dfr(votes_list, flatten_dfc)

    # Add Vote tallies
    vote_rolls_df = votes_df %>% 
      mutate(vote_roll = map(url, parse_vote_roll, 
                             logger = logger, 
                             bill_type = bill_df$billType,
                             bill_num = bill_df$billNumber),
             roll_found = map_lgl(vote_roll, ~(nrow(.) > 0))) %>% 
      janitor::clean_names()
    
    bill_df$votes = list(vote_rolls_df)
  } else {
    bill_df$votes = list(tibble())
  }
  
  # Actions ---------
  log_debug(logger, 
           bill_type = bill_df$billType,
           bill_num = bill_df$billNumber,
           "Parsing actions")
  
  bill_actions = xml_find_all(bill_xml, "//bill/actions/item")
  
  if("actions" %in% nested_attributes && length(bill_actions)>0){

    bill_action_counts = as_list(xml_find_all(bill_xml, "//bill/actions/*[not(self::item)]")) %>% 
      map_dfc(flatten_dfc) %>% 
      rename_with(.cols = everything(), ~str_c("actions_", .)) %>% 
      pivot_longer(everything(), names_to = "action", names_prefix = "actions_", values_to = "count")
    
    # Coerce nodes to list
    actions_df = as_list(bill_actions) %>% 
      map_dfr(parse_action) %>% 
      type_convert(col_types = col_specs$actions) %>% 
      janitor::clean_names()
    
    bill_df$actions = list(actions_df)
    
    bill_df$action_counts = list(type_convert(bill_action_counts,
                                              col_types = cols(action = col_character(), count = col_integer())))
  } else {
    bill_df$actions = list(tibble())
  }
  
  # Sponsors ---------
  log_debug(logger, 
           bill_type = bill_df$billType,
           bill_num = bill_df$billNumber,
           "Parsing sponsors")
  
  bill_sponsors = xml_find_all(bill_xml, "//bill/sponsors/item")
  
  if("sponsors" %in% nested_attributes && length(bill_sponsors)>0){

    # Coerce nodes to list
    sponsors_df = as_list(bill_sponsors) %>% 
      map_dfr(parse_sponsor) %>% 
      janitor::clean_names()
    
    bill_df$sponsors = list(sponsors_df)
  } else {
    bill_df$sponsors = list(tibble())
  }
  
  
  # Cosponsors ---------
  log_debug(logger, 
           bill_type = bill_df$billType,
           bill_num = bill_df$billNumber,
           "Parsing cosponsors")
  
  bill_cosponsors = xml_find_all(bill_xml, "//bill/cosponsors/item")
  
  if("cosponsors" %in% nested_attributes && length(bill_cosponsors)>0){
    
    # Coerce nodes to list
    cosponsors_df = as_list(bill_cosponsors) %>% 
      map_dfr(parse_sponsor, role = "cosponsor") %>% 
      janitor::clean_names()
    
    bill_df$cosponsors = list(type_convert(cosponsors_df, col_types = col_specs$cosponsors))
  } else {
    bill_df$cosponsors = list(tibble())
  }
  
  finished_df = as_tibble(bill_df) %>% 
    janitor::clean_names() %>% 
    # Combine bill type and number to create an ID
    unite(bill_id, bill_type, bill_number, sep = "-", remove = F)
  
  log_info(logger, 
            bill_type = bill_df$billType,
            bill_num = bill_df$billNumber,
            "Complete")
  
  mutate(finished_df,
         across(ends_with("date"), as_datetime),
         actions = map(actions, mutate, 
                       action_type = factor(action_type, 
                                            levels = c("IntroReferral", "Committee", "Floor", 
                                                       "Discharge", "President", "BecameLaw"), 
                                            ordered = T))
         )
}

# Source

This data source is derived from govinfo's `BILLSTATUS` bulk data, which can be found [here](https://www.govinfo.gov/bulkdata). [^1]: Meaning for characteristics which cannot be represented in a single row tidily, they are nested in a list column. This gives the user the flexibility to change the level of observation depending on their analytical aim.
